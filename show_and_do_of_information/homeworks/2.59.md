~~~
#include<stdio.h>

typedef unsigned char *byte_pointer;

void show_bytes(byte_pointer s1,byte_pointer s2,size_t i){
	for(int j=i-1;j>0;j--){
		printf("%02x",s1[j]);
	}
	printf("%02x",s2[0]);
}

int main (){
	
	int x,y;
	//scanf("%d %d",&x,&y);
	x=0x76543210;
	y=0x89abcdef;
	show_bytes((byte_pointer) &x,(byte_pointer) &y,sizeof(int));
	
	return 0;
}
~~~
对show_bytes略作修改即可实现拼接。
